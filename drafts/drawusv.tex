% USV model A: hull only
% usage: \drawusv{<shift coordinate>}{<rotation angle>}{<scale>}
% -------------------------------------------------
\newcommand{\drawusv}[3]{% 位置、旋转、缩放倍数
	\begin{scope}[shift={#1}, rotate=#2, scale=#3]
		% hull fill
		\fill[gray!35]
		(0,1.15)
		.. controls (0.28,0.95) and (0.42,0.45) .. (0.40,-0.10)
		.. controls (0.38,-0.48) and (0.30,-0.78) .. (0.24,-0.90)
		-- (-0.24,-0.90)
		.. controls (-0.30,-0.78) and (-0.38,-0.48) .. (-0.40,-0.10)
		.. controls (-0.42,0.45) and (-0.28,0.95) .. cycle;
		
		% hull outline
		\draw[thick]
		(0,1.15)
		.. controls (0.28,0.95) and (0.42,0.45) .. (0.40,-0.10)
		.. controls (0.38,-0.48) and (0.30,-0.78) .. (0.24,-0.90)
		-- (-0.24,-0.90)
		.. controls (-0.30,-0.78) and (-0.38,-0.48) .. (-0.40,-0.10)
		.. controls (-0.42,0.45) and (-0.28,0.95) .. cycle;
		
%		% centerline
%		\draw[thin] (0,0.92) -- (0,-0.74);
	\end{scope}%
}
