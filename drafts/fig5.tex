\documentclass[tikz,border=6pt]{standalone}
\usepackage{ctex}
\usepackage{amsmath}
\usetikzlibrary{arrows.meta,calc,angles,quotes}

\begin{document}
\begin{tikzpicture}[
    >=Stealth,
    font=\small,
    line join=round,
    line cap=round
]

% ======================================
% 参数
% ======================================
\def\gammaPath{18}   % 路径切向角
\def\ey{1.5}         % 横向误差
\def\psiCur{42}      % 当前航向

% ======================================
% 全局坐标系
% ======================================
\coordinate (O) at (0,0);
\draw[->, thick] (O) -- (8.2,0) node[below right] {$X$};
\draw[->, thick] (O) -- (0,5.2) node[left] {$Y$};
\node[below left] at (O) {$O$};

% ======================================
% 路径投影点 Pr
% ======================================
\coordinate (Pr) at (3.4,1.8);

% % 路径线
% \draw[thick] ($(Pr)+(-4.3*cos(\gammaPath),-4.3*sin(\gammaPath))$)
%           -- ($(Pr)+(4.6*cos(\gammaPath),4.6*sin(\gammaPath))$)
%           node[above right] {Reference path};

% 当前艇位置 P
\coordinate (P) at ($(Pr)+({\ey*(-sin(\gammaPath))},{\ey*(cos(\gammaPath))})$);

% 点标记
\fill (Pr) circle (1.1pt);
\node[below right=1pt] at (Pr) {$P_r$};

\fill (P) circle (1.1pt);
\node[above left=1pt] at (P) {$P$};

% 横向误差
\draw[dashed] (P) -- (Pr);
\node[left] at ($(P)!0.5!(Pr)$) {$e_y$};

% 路径切向方向
\coordinate (Tend) at ($(Pr)+({1.7*cos(\gammaPath)},{1.7*sin(\gammaPath)})$);
\draw[->, thick] (Pr) -- (Tend);

% 当前航向方向
\coordinate (PsiEnd) at ($(P)+({1.9*cos(\psiCur)},{1.9*sin(\psiCur)})$);
\draw[->, thick] (P) -- (PsiEnd);

% 小船示意
\begin{scope}[shift={(P)}, rotate=\psiCur]
    \def\L{0.9}
    \def\W{0.36}
    \draw[thick]
        (-0.42*\L,-0.5*\W)
        -- (0.18*\L,-0.5*\W)
        -- (0.52*\L,0)
        -- (0.18*\L,0.5*\W)
        -- (-0.42*\L,0.5*\W)
        -- cycle;
    \draw[dashed] (-0.38*\L,0) -- (0.46*\L,0);
\end{scope}

% gamma_p 角
\coordinate (XrefPr) at ($(Pr)+(1.3,0)$);
\draw pic[
    draw,
    ->,
    angle radius=8mm,
    angle eccentricity=1.18,
    "$\gamma_p$"
] {angle = XrefPr--Pr--Tend};

% psi 角
\coordinate (XrefP) at ($(P)+(1.2,0)$);
\draw pic[
    draw,
    ->,
    angle radius=8mm,
    angle eccentricity=1.2,
    "$\psi$"
] {angle = XrefP--P--PsiEnd};

% e_psi 角：路径切向方向与当前航向之间的夹角
\coordinate (TdirFromP) at ($(P)+({1.5*cos(\gammaPath)},{1.5*sin(\gammaPath)})$);
\draw[dashed] (P) -- (TdirFromP);

\draw pic[
    draw,
    ->,
    angle radius=13mm,
    angle eccentricity=1.08,
    "$e_\psi$"
] {angle = TdirFromP--P--PsiEnd};

\end{tikzpicture}
\end{document}
