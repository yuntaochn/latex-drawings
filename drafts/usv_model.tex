\documentclass[tikz,border=8pt]{standalone}
\usepackage[UTF8]{ctex}
\usepackage{amsmath}
\usetikzlibrary{arrows.meta,positioning,fit,backgrounds,calc,shapes.geometric}

\definecolor{cblue}{RGB}{232,242,255}
\definecolor{corange}{RGB}{255,242,227}
\definecolor{cgreen}{RGB}{234,247,237}
\definecolor{cgray}{RGB}{245,246,248}
\definecolor{cline}{RGB}{80,92,110}
\definecolor{ctext}{RGB}{35,42,52}

\tikzset{
	base/.style={draw=cline, rounded corners=2.5mm, thick, align=center, text=ctext, minimum height=10mm},
	block/.style={base, fill=cgray, minimum width=28mm, inner sep=5pt},
	kin/.style={base, fill=cblue, minimum width=34mm, inner sep=6pt},
	dyn/.style={base, fill=corange, minimum width=34mm, inner sep=6pt},
	io/.style={base, fill=cgreen, minimum width=26mm, inner sep=5pt},
	groupbox/.style={draw=cline!85, rounded corners=3mm, dashed, inner sep=7pt, thick},
	arrow/.style={-{Latex[length=3.2mm,width=2.2mm]}, thick, draw=cline},
	fb/.style={-{Latex[length=2.8mm,width=2.0mm]}, thick, draw=cline!80},
	grouplabel/.style={font=\bfseries\small, text=ctext, fill=white, inner sep=2pt}
}

\begin{document}
	\begin{tikzpicture}[node distance=8mm and 7mm, font=\small]
		
		% Main chain
		\node[io]   (cmd) {控制输入\\$u_c$};
		\node[block, right=of cmd] (map) {执行器/控制分配\\$u_c \rightarrow \tau$};
		\node[dyn, right=of map]   (dyn) {动力学模型\\$M\dot{\nu}+C(\nu)\nu+D(\nu)\nu=\tau+d$};
		\node[io, right=of dyn]    (vel) {速度状态\\$\nu=[u,v,r]^T$};
		\node[kin, right=of vel]   (kin) {运动学模型\\$\dot{\eta}=J(\psi)\nu$};
		\node[io, right=of kin]    (pose) {位姿状态\\$\eta=[x,y,\psi]^T$};
		
		\draw[arrow] (cmd) -- (map);
		\draw[arrow] (map) -- node[above=1.5mm, font=\small] {合力/力矩 $\tau$} (dyn);
		\draw[arrow] (dyn) -- node[above=1.5mm, font=\small] {积分} (vel);
		\draw[arrow] (vel) -- node[above=1.5mm, font=\small] {坐标变换} (kin);
		\draw[arrow] (kin) -- node[above=1.5mm, font=\small] {积分} (pose);
		
		% Disturbance
		\node[block, above=11mm of dyn, minimum width=28mm] (dist) {外界扰动\\风/浪/流 $d$};
		\draw[arrow] (dist) -- (dyn);
		
		% Feedback control path
		\node[block, below=16mm of dyn, minimum width=34mm] (ctrl) {导引/控制器\\误差计算与控制律};
		\draw[fb] (pose.south) |- ++(0,-11mm) -| node[pos=0.30, below=2mm] {位置/航向反馈} (ctrl.east);
		\draw[fb] (vel.south) -- ++(0,-7mm) -| node[pos=0.25, below=2mm] {速度反馈} (ctrl.east);
		\draw[fb] (ctrl.west) -| ++(-18mm,0) |- node[pos=0.25, below=2mm] {期望指令} (cmd.south);
		
		% Group boxes
		\begin{pgfonlayer}{background}
			\node[groupbox, fit=(dyn)(vel), label={[grouplabel]north:动力学层}] {};
			\node[groupbox, fit=(kin)(pose), label={[grouplabel]north:运动学层}] {};
			\node[groupbox, fit=(cmd)(map)(dyn)(vel)(kin)(pose), inner sep=10pt] {};
		\end{pgfonlayer}
		
		% Bottom summary
		\node[align=center, text=ctext, below=34mm of dyn, font=\small] (sum) {
			\textbf{作用链：}控制输入 $\rightarrow$ 广义力/力矩 $\rightarrow$ 速度变化（动力学） $\rightarrow$ 位姿变化（运动学）\\
			\textbf{核心关系：}动力学回答“\emph{力如何改变速度}”，运动学回答“\emph{速度如何改变位置与姿态}”
		};
		
	\end{tikzpicture}
\end{document}
