\documentclass[tikz,border=5pt]{standalone}
\usetikzlibrary{arrows.meta,calc,angles,quotes}

\begin{document}
	\begin{tikzpicture}[>=Latex]
		
		% =================================================
		% 1. 全局坐标系
		% =================================================
		\draw[->,thick] (0,0) -- (18,0) node[below] {X(East)};
		\draw[->,thick] (0,0) -- (0,15) node[left]  {Y(North)};
		
		% =================================================
		% 2. 路径、当前位置、投影点、LOS点
		% =================================================
		\coordinate (Pk)  at (2,5);
		\coordinate (Pkp) at (16,13.17);
		\coordinate (P)   at (7.5,1);
		
		% 路径
		\draw[thick] (Pk) -- (Pkp);
		\fill (Pk)  circle (1.2pt);
		\fill (Pkp) circle (1.2pt);
		\node[below left]  at (Pk)  {$p_k$};
		\node[above right] at (Pkp) {$p_{k+1}$};
		
		% 自动正交投影点
		\coordinate (Pproj) at ($(Pk)!(P)!(Pkp)$);
		\fill (Pproj) circle (1.2pt);
		\node[below] at (Pproj) {$p_\perp$};
		
		% 自动生成 LOS 点：从投影点沿路径方向前移
		\coordinate (Plos) at ($(Pproj)!0.72!(Pkp)$);
		\fill (Plos) circle (1.2pt);
		\node[above] at (Plos) {$p_{\mathrm{LOS}}$};
		
        % 横向误差 e
        \draw[thick] (P) -- (Pproj)
        node[pos=0.52, left=2pt] {$e_y$};
        % node[pos=0.52, sloped, above=2pt] {$e$};

        % 前视距离 Delta
        \draw[<->] (Pproj) -- (Plos)
        % node[pos=0.55, sloped, above=2pt] {$\Delta$};
        node[pos=0.55, above=2pt] {$\Delta$};

		% LOS 向量
		\draw[->,thick,dashed] (P) -- (Plos) node[midway,above right] {LOS};
		
		% =================================================
		% 3. 局部 North 参考线（一定要画出来）
		% =================================================
		\coordinate (NorthRefP)  at ($(P)+(0,3.2)$);
		\coordinate (NorthRefPk) at ($(Pk)+(0,2.6)$);
		
		\draw[dashed] (P)  -- (NorthRefP);
		\draw[dashed] (Pk) -- (NorthRefPk);
		
		% =================================================
		% 4. 自动计算方向角（内部仍按 East-based）
		%    但图上的角统一“按 North 顺时针”解释
		% =================================================
		
		% 路径方向 alpha_k
		\pgfmathanglebetweenpoints{\pgfpointanchor{Pk}{center}}{\pgfpointanchor{Pkp}{center}}
		\pgfmathsetmacro{\alphaE}{\pgfmathresult}
		
		% LOS 对应的期望航迹角 chi_d
		\pgfmathanglebetweenpoints{\pgfpointanchor{P}{center}}{\pgfpointanchor{Plos}{center}}
		\pgfmathsetmacro{\chiDE}{\pgfmathresult}
		
		% 你可以手调这两个偏差
		\pgfmathsetmacro{\chiE}{\chiDE + 8}   % 实际航迹角 chi
		\pgfmathsetmacro{\psiE}{\chiE  + 10}  % 实际航向角 psi
		% 这样 beta = psi - chi 在 East-based 下为正，
		% 对应图上“从 psi 到 chi 的顺时针小角”
        % 实际航迹 $\chi$ 离 $\chi_d$ 太远，就把 `+8` 改小；
        % 侧滑角 $\beta$ 太大，就把 `+10` 改小。
        
		% =================================================
		% 5. 构造方向辅助点
		% =================================================
		\coordinate (ChiPoint) at ($(P)+({4.0*cos(\chiE)},{4.0*sin(\chiE)})$);
		\coordinate (PsiPoint) at ($(P)+({3.2*cos(\psiE)},{3.2*sin(\psiE)})$);
		
		% =================================================
		% 6. 画 USV（朝向 psi）
		% =================================================
		\def\L{1.6}
		\def\B{0.6}
		\def\Lrect{1.0}
		
		\begin{scope}[shift={(P)}, rotate=\psiE]
			\coordinate (SternL) at (-\Lrect/2, -\B/2);
			\coordinate (SternR) at (-\Lrect/2,  \B/2);
			\coordinate (BowR)   at ( \Lrect/2,  \B/2);
			\coordinate (BowTip) at ( \L/2,      0);
			\coordinate (BowL)   at ( \Lrect/2, -\B/2);
			
			% 只画轮廓，不填充
			\draw[thick] (SternL) -- (SternR) -- (BowR) -- (BowTip) -- (BowL) -- cycle;
			
			% Body-fixed frame
			\draw[->,dashed] (0,0) -- (3,0) node[above] {$x_B$};
			\draw[->,dashed] (0,0) -- (0,-1) node[above] {$y_B$};
		\end{scope}
		
		\node[above left] at (P) {$p$};
		
		% 实际速度方向 U（对应 chi）
		\draw[->,dashed] (P) -- (ChiPoint) node[above right] {$U$};
		
		% =================================================
		% 7. 自动角度标注（全部按 North 顺时针理解）
		%    关键：用 Target--Vertex--NorthRef，并配合 <- 
		% =================================================
		
		% alpha_k：从 North 到路径方向
		\draw pic[
		draw,
		<-,
		angle radius=10mm,
		angle eccentricity=1.2,
		"$\alpha_k$"
		] {angle = Pkp--Pk--NorthRefPk};
		
		% psi：从 North 到航向
		\draw pic[
		draw,
		<-,
		angle radius=13mm,
		angle eccentricity=1.18,
		"$\psi$"
		] {angle = PsiPoint--P--NorthRefP};
		
		% chi_d：从 North 到期望航迹（LOS方向）
		\draw pic[
		draw,
		<-,
		angle radius=18mm,
		angle eccentricity=1.15,
		"$\chi_d$"
		] {angle = Plos--P--NorthRefP};
		
		% chi：从 North 到实际航迹
		\draw pic[
		draw,
		<-,
		angle radius=23mm,
		angle eccentricity=1.15,
		"$\chi$"
		] {angle = ChiPoint--P--NorthRefP};
		
		% beta：从 psi 到 chi（顺时针小角）
		\draw pic[
		draw,
		<-,
		angle radius=14mm,
		angle eccentricity=1.15,
		"$\beta$"
		] {angle = ChiPoint--P--PsiPoint};
		
	\end{tikzpicture}
\end{document}
